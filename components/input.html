# ✨双向绑定

message:双向绑定

通过v-model属性实现双向绑定

<template>
    <div>
        <wb-input v-model="message"></wb-input>
        <div style="margin-top: 20px">message:{{message}}</div>
    </div>
</template>
<script>
    export default {
        data() {
            return {
                message: '双向绑定',
            }
        }
    }
</script>

# ✨事件绑定

<template>
    <div>
        <wb-input value="事件绑定" @change="callback"></wb-input>
        <wb-input value="事件绑定" @input="callback2"></wb-input>
    </div>
</template>
<script>
    export default {
        methods:{
            callback(value){
                console.log(value);
            },
            callback2(value){
                console.log(value);
            }
        }
    }
</script>
